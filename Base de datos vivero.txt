
DROP TABLE conductor cascade;
DROP TABLE cliente cascade;
DROP TABLE usuario cascade;
DROP TABLE planta cascade;
DROP TABLE salida cascade;
DROP TABLE detalle_ingreso cascade;
DROP TABLE ingreso_planta cascade;
DROP TABLE detalle_salida cascade;
DROP TABLE reserva cascade;


CREATE TABLE Conductor (id_conductor int not null PRIMARY KEY, nombre varchar(40) not null,apellido varchar(30) not null,celular bigint, estado varchar(15) not null );

CREATE TABLE Cliente (id_cliente  varchar(14) not null PRIMARY KEY, nombre varchar(30) not null,celular bigint not null );

CREATE TABLE Usuario (usuario varchar (20) not null PRIMARY KEY, contraseña varchar(20) not null, cargo varchar(10), nombre_usuario varchar(30) not null, apellido varchar (30) not null, celular bigint not null,Direccion varchar(30), estado varchar(10) not null);

CREATE TABLE Planta (codigo_planta SERIAL not null   PRIMARY KEY, nombre_vulgar varchar (40) not null, nombre_cientifico varchar(50) not null,familia varchar (40) ,msnm varchar(40) not null );                                                                            

CREATE TABLE Salida (codigo_salida int not null PRIMARY KEY, numero_orden varchar(12) not null, motivo varchar(15) not null, fecha date not null, parcial int, observacion varchar(180), destino varchar(30), convenio varchar(50), corregimiento varchar(20), carga varchar(25),predio varchar(30) ,id_cliente varchar(30), id_conductor int, placa varchar(7), id_conductor_temp varchar(10), nombre_conductor_temp varchar(40), apellido_conductor_temp varchar(30), celular_conductor_temp varchar(10), FOREIGN KEY (id_conductor) REFERENCES conductor (id_conductor));

CREATE TABLE Ingreso_Planta (codigo_ingreso varchar(18) not null PRIMARY KEY, codigo_planta int not null, tamaño_bolsa varchar(10) not null, rango varchar(20) ,stock int not null, disponibles int not null, reservadas int not null, FOREIGN KEY (codigo_planta) REFERENCES planta(codigo_planta));

CREATE TABLE Detalle_Ingreso (identificador SERIAL not null,codigo_planta int not null, cantidad int not null, fecha date not null, codigo_ingreso varchar(18) not null , FOREIGN KEY (codigo_planta) REFERENCES planta(codigo_planta),  FOREIGN KEY (codigo_ingreso) REFERENCES ingreso_planta (codigo_ingreso) );

CREATE TABLE Detalle_Salida (codigo_salida int not null, codigo_ingreso varchar(18) not null, cantidad int not null, pedidas int, tamaño_destino varchar(20), costo bigint, total bigint, FOREIGN KEY (codigo_ingreso) REFERENCES Ingreso_planta (codigo_ingreso), FOREIGN KEY (codigo_salida) REFERENCES salida (codigo_salida)   );

CREATE TABLE Reserva (codigo_reserva serial not null, id_cliente varchar(14) not null, codigo_ingreso VARCHAR(18) not null, tamaño_bolsa varchar(10), rango varchar(20), cantidad int not null, costo_reserva bigint not null, fecha_apertura date, fecha_cierre date, FOREIGN KEY (id_cliente) REFERENCES Cliente(id_cliente), FOREIGN KEY (codigo_ingreso) REFERENCES Ingreso_Planta(codigo_ingreso) );
